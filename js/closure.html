闭包(closure)

目标：理解 javascript中的闭包(closure)


在函数内保持数据的特性被称为闭包 —— 《javascript语言精粹和编程实战 第二版》
《javascript语言精粹和编程实战 第二版》 P209 
		var set , get;
		function MyFunc(){
			//初值
			var value = 100;
			//内部函数
			function set_value(v){
				value = v;
			}
			function get_value(){
				return value;
			}
			//将内部函数公布到全局
			set = set_value;
			get = get_value;
		}
		//测试一
		//输出100
		MyFunc();
		alert(get());

		//测试二
		//输出300
		set(300);
		alert(get());
